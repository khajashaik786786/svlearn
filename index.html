apb_if.sv — Interface with clocking blocks
// ═══ APB Interface — Interface Iris's masterpiece ═══ // One interface replaces dozens of port connections interface apb_if(input logic clk, rst_n); // APB signals logic [11:0] paddr; logic [31:0] pwdata, prdata; logic pwrite, psel, penable, pready, pslverr; logic [3:0] pstrb; // Clocking block for DRIVER — eliminates setup/hold races! clocking drv_cb @(posedge clk); default input #1step output #1; output paddr, pwdata, pwrite, psel, penable, pstrb; input pready, prdata, pslverr; endclocking // Clocking block for MONITOR — sample just before edge clocking mon_cb @(posedge clk); default input #1step; input paddr, pwdata, pwrite, psel, penable, pready, prdata; endclocking // Modports: each component sees only what it needs modport drv_mp (clocking drv_cb, input rst_n); modport mon_mp (clocking mon_cb, input rst_n); modport dut_mp (input paddr, pwdata, pwrite, psel, penable, pstrb, clk, rst_n, output pready, prdata, pslverr); // Assertions live INSIDE the interface — always checking! property p_sel_before_en; @(posedge clk) disable iff(!rst_n) penable |-> psel; // penable can only be high if psel is high endproperty assert property(p_sel_before_en) else $error("APB VIOLATION: penable without psel!"); endinterface
apb_transaction.sv — Data class with constraints
// ═══ TRANSACTION CLASS — Dr. Byte + Rex Randon collaborated ═══ class apb_transaction; // Two-state rand fields (Dr. Byte: "bit is faster than logic!") rand bit [11:0] addr; rand bit [31:0] data; rand bit write; // 0=read, 1=write rand bit [3:0] strb; // byte enables // Constraints (Rex: "guided randomness beats directed every time!") constraint c_addr_align { addr[1:0] == 2'b00; // word-aligned only addr inside {[12'h000:12'hFFF]}; } constraint c_strb_valid { write == 1 -> strb != 4'h0; // writes must have byte enables write == 0 -> strb == 4'hF; // reads: full strobe } constraint c_write_bias { write dist { 1 := 70, 0 := 30 }; // 70% writes, 30% reads } // Constructor function new(); endfunction // Print helper function string sprint(); return $sformatf("APB: addr=%3h data=%8h %s strb=%b", addr, data, write?"WR":"RD", strb); endfunction // Deep copy function apb_transaction copy(); apb_transaction c = new(); c.addr = addr; c.data = data; c.write = write; c.strb = strb; return c; endfunction endclass
apb_generator.sv — Stimulus engine
// ═══ GENERATOR — Rex Randon's stimulus factory ═══ class apb_generator; mailbox#(apb_transaction) gen2drv; // send to driver via mailbox int num_txns; event done; function new(mailbox#(apb_transaction) mb, int n=20); gen2drv = mb; num_txns = n; endfunction task run(); apb_transaction tr; // 1. Random transactions repeat(num_txns) begin tr = new(); assert(tr.randomize()) else $fatal(1, "Randomization failed!"); gen2drv.put(tr.copy()); end // 2. Directed: test boundary addresses foreach (bit[11:0] ba[12'h000, 12'hFFC, 12'h800]) begin tr = new(); tr.randomize() with { addr == ba; write == 1; }; gen2drv.put(tr.copy()); end ->done; // signal that all transactions are queued $display("[GEN] %0d transactions generated", num_txns+3); endtask endclass
apb_driver.sv — Protocol driver via clocking block
// ═══ DRIVER — Interface Iris's bus engine ═══ class apb_driver; virtual apb_if vif; // virtual interface handle mailbox#(apb_transaction) gen2drv; int txn_count = 0; function new(virtual apb_if vif_in, mailbox#(apb_transaction) mb); vif = vif_in; gen2drv = mb; endfunction task reset_bus(); @(posedge vif.clk); vif.drv_cb.psel <= 0; vif.drv_cb.penable <= 0; vif.drv_cb.paddr <= 0; vif.drv_cb.pwdata <= 0; vif.drv_cb.pwrite <= 0; vif.drv_cb.pstrb <= 4'hF; endtask task run(); apb_transaction tr; reset_bus(); @(posedge vif.rst_n); // wait for reset release @(vif.drv_cb); forever begin gen2drv.get(tr); // blocks until transaction ready drive_txn(tr); txn_count++; end endtask task drive_txn(apb_transaction tr); // SETUP phase @(vif.drv_cb); vif.drv_cb.paddr <= tr.addr; vif.drv_cb.pwdata <= tr.data; vif.drv_cb.pwrite <= tr.write; vif.drv_cb.pstrb <= tr.strb; vif.drv_cb.psel <= 1; vif.drv_cb.penable <= 0; // ACCESS phase @(vif.drv_cb); vif.drv_cb.penable <= 1; // Wait for DUT ready @(vif.drv_cb iff vif.pready === 1); // End transfer vif.drv_cb.psel <= 0; vif.drv_cb.penable <= 0; endtask endclass
apb_monitor.sv — Silent observer
// ═══ MONITOR — never drives, only observes ═══ class apb_monitor; virtual apb_if vif; mailbox#(apb_transaction) mon2sb; // send captured txns to scoreboard mailbox#(apb_transaction) mon2cov; // also feed coverage collector int pkt_count = 0; function new(virtual apb_if vif_in, mailbox#(apb_transaction) sb_mb, mailbox#(apb_transaction) cov_mb); vif = vif_in; mon2sb = sb_mb; mon2cov = cov_mb; endfunction task run(); apb_transaction tr; forever begin // Watch for: sel && enable && ready (complete APB transfer) @(vif.mon_cb iff (vif.mon_cb.psel && vif.mon_cb.penable && vif.pready)); tr = new(); tr.addr = vif.mon_cb.paddr; tr.write = vif.mon_cb.pwrite; tr.strb = 4'hF; tr.data = vif.mon_cb.pwrite ? vif.mon_cb.pwdata : vif.prdata; mon2sb.put(tr.copy()); // to scoreboard mon2cov.put(tr.copy()); // to coverage pkt_count++; $display("[MON] Captured: %s", tr.sprint()); end endtask endclass
apb_scoreboard.sv — Reference model + checker
// ═══ SCOREBOARD — OOP Omar's reference model + checker ═══ class apb_scoreboard; // Array Anna: "associative array = perfect register model!" bit [31:0] ref_model[bit[11:0]]; // addr → expected value mailbox#(apb_transaction) mon2sb; int pass_cnt = 0, fail_cnt = 0; function new(mailbox#(apb_transaction) mb); mon2sb = mb; endfunction task run(); apb_transaction tr; forever begin mon2sb.get(tr); if (tr.write) begin ref_model[tr.addr] = tr.data; // update reference end else begin if (ref_model.exists(tr.addr)) begin if (tr.data === ref_model[tr.addr]) begin pass_cnt++; $display("[SB] PASS: addr=%3h", tr.addr); end else begin fail_cnt++; $error("[SB] FAIL addr=%3h exp=%8h got=%8h", tr.addr, ref_model[tr.addr], tr.data); end end end end endtask function void report(); $display("═══ SCOREBOARD: PASS=%0d FAIL=%0d ═══", pass_cnt, fail_cnt); if (fail_cnt > 0) $error("TEST FAILED!"); else $display("✓ TEST PASSED!"); endfunction endclass
apb_coverage.sv — Functional coverage
// ═══ COVERAGE — Rex Randon tracks what's been tested ═══ class apb_coverage; mailbox#(apb_transaction) mon2cov; apb_transaction curr_tr; // Covergroup: your verification contract! covergroup apb_cg; cp_addr: coverpoint curr_tr.addr { bins low_regs = {[12'h000:12'h0FF]}; bins mid_regs = {[12'h100:12'h7FF]}; bins high_regs = {[12'h800:12'hFFF]}; } cp_rw: coverpoint curr_tr.write { bins read_op = {1'b0}; bins write_op = {1'b1}; } cp_strb: coverpoint curr_tr.strb { bins full = {4'hF}; bins upper = {4'hC}; bins lower = {4'h3}; bins single = {4'h1,4'h2,4'h4,4'h8}; } // Cross coverage: addr range × read/write = 6 cross bins cx_addr_rw: cross cp_addr, cp_rw; endgroup function new(mailbox#(apb_transaction) mb); mon2cov = mb; apb_cg = new(); endfunction task run(); forever begin mon2cov.get(curr_tr); apb_cg.sample(); end endtask function void report(); $display("Coverage: %.1f%%", apb_cg.get_coverage()); endfunction endclass
apb_env.sv — Environment (wires everything together)
// ═══ ENVIRONMENT — The testbench orchestrator ═══ class apb_env; // Components (OOP Omar: "composition over everything!") apb_generator gen; apb_driver drv; apb_monitor mon; apb_scoreboard sb; apb_coverage cov; // Mailboxes for inter-component communication mailbox#(apb_transaction) gen2drv; // generator → driver mailbox#(apb_transaction) mon2sb; // monitor → scoreboard mailbox#(apb_transaction) mon2cov; // monitor → coverage function new(virtual apb_if vif, int n_txns=30); gen2drv = new(); mon2sb = new(); mon2cov = new(); gen = new(gen2drv, n_txns); drv = new(vif, gen2drv); mon = new(vif, mon2sb, mon2cov); sb = new(mon2sb); cov = new(mon2cov); endfunction task run(); // Forky Fiona: "fork all components — they run in parallel!" fork gen.run(); drv.run(); mon.run(); sb.run(); cov.run(); join_any // gen finishes first (finite transactions) @(gen.done); // wait for generator event #100; // drain in-flight transactions $finish; endtask function void report(); sb.report(); cov.report(); endfunction endclass
tb_top.sv — Top-level module (the entry point)
// ═══ TOP LEVEL — everything wired together ═══ `include "apb_if.sv" `include "apb_transaction.sv" `include "apb_generator.sv" `include "apb_driver.sv" `include "apb_monitor.sv" `include "apb_scoreboard.sv" `include "apb_coverage.sv" `include "apb_env.sv" module tb_top; // Clock & Reset logic clk = 0; logic rst_n = 0; always #5 clk = ~clk; // 100 MHz clock // Interface instance (Interface Iris's masterpiece!) apb_if dut_if(.clk(clk), .rst_n(rst_n)); // DUT instance apb_slave_dut dut( .clk (clk), .rst_n (rst_n), .paddr (dut_if.paddr), .pwdata (dut_if.pwdata), .pwrite (dut_if.pwrite), .psel (dut_if.psel), .penable(dut_if.penable), .pstrb (dut_if.pstrb), .pready (dut_if.pready), .prdata (dut_if.prdata), .pslverr(dut_if.pslverr) ); apb_env env; initial begin // Reset sequence rst_n = 0; repeat(5) @(posedge clk); rst_n = 1; // Create environment and run! env = new(dut_if, 50); // 50 transactions env.run(); env.report(); $finish; end // Waveform dump initial begin $dumpfile("apb_tb.vcd"); $dumpvars(0, tb_top); end endmodule
