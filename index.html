// ═══ TRANSACTION ITEM — the data object passed through the testbench ═══ // Dr. Byte designed the fields; Randon Rex added the constraints class apb_transaction extends uvm_sequence_item; `uvm_object_utils_begin(apb_transaction) `uvm_field_int(addr, UVM_ALL_ON) `uvm_field_int(data, UVM_ALL_ON) `uvm_field_int(write, UVM_ALL_ON) `uvm_field_int(strb, UVM_ALL_ON) `uvm_object_utils_end // ── Two-state rand fields (Dr. Byte: "bit is 2x faster than logic!") rand bit [11:0] addr; // 12-bit APB address rand bit [31:0] data; // 32-bit data rand bit write; // 0=read, 1=write rand bit [3:0] strb; // byte strobe // ── Constraints (Randon Rex: "guided randomness wins every time!") constraint c_addr_align { addr[1:0] == 2'b00; // word-aligned addr inside {[12'h000:12'hFFF]}; } constraint c_strb_write { write == 1 -> strb != 4'h0; // writes need valid strobes write == 0 -> strb == 4'hF; // reads: full strobe } constraint c_wr_bias { write dist { 1 := 70, 0 := 30 }; // 70% writes, 30% reads } function new(string name = "apb_transaction"); super.new(name); endfunction endclass

// ═══ SEQUENCE — Randon Rex orchestrates stimulus scenarios ═══ // Base sequence (OOP Omar: "always have a base class!") virtual class apb_seq_base extends uvm_sequence#(apb_transaction); `uvm_object_utils(apb_seq_base) function new(string name="apb_seq_base"); super.new(name); endfunction endclass // Write burst sequence class apb_write_seq extends apb_seq_base; `uvm_object_utils(apb_write_seq) rand int unsigned burst_len = 10; constraint c_len { burst_len inside {[5:32]}; } task body(); apb_transaction tr; `uvm_info("SEQ", $sformatf("Write burst: %0d transactions", burst_len), UVM_LOW) repeat(burst_len) begin tr = apb_transaction::type_id::create("tr"); start_item(tr); if(!tr.randomize() with { write == 1; }) `uvm_fatal("SEQ", "Randomization failed!") finish_item(tr); end endtask endclass // Mixed read-write sequence class apb_mixed_seq extends apb_seq_base; `uvm_object_utils(apb_mixed_seq) task body(); apb_write_seq wr_seq; apb_transaction rd_tr; // First: write phase wr_seq = apb_write_seq::type_id::create("wr_seq"); wr_seq.start(m_sequencer); // Then: read back (Forky Fiona: "no fork here — sequential is safer!") repeat(5) begin rd_tr = apb_transaction::type_id::create("rd"); start_item(rd_tr); rd_tr.randomize() with { write == 0; }; finish_item(rd_tr); end endtask endclass

// ═══ DRIVER — Interface Iris's bus-driving engine ═══ // Uses clocking block to avoid races; virtual interface for DUT access class apb_driver extends uvm_driver#(apb_transaction); `uvm_component_utils(apb_driver) virtual apb_if vif; // Interface Iris: "virtual interface is the gateway!" function void build_phase(uvm_phase phase); super.build_phase(phase); if(!uvm_config_db#(virtual apb_if)::get(this,"","vif",vif)) `uvm_fatal("DRV","No virtual interface!") endfunction task run_phase(uvm_phase phase); apb_transaction req; vif.cb.psel <= 0; // initialize bus to idle vif.cb.penable <= 0; forever begin seq_item_port.get_next_item(req); // blocks until item ready drive_transaction(req); seq_item_port.item_done(); // release Wizard's finish_item() end endtask task drive_transaction(apb_transaction tr); // APB 3-phase protocol: SETUP → ACCESS → complete @(vif.cb); // sync to clocking block edge vif.cb.paddr <= tr.addr; vif.cb.pwdata <= tr.data; vif.cb.pwrite <= tr.write; vif.cb.pstrb <= tr.strb; vif.cb.psel <= 1; // SETUP phase @(vif.cb); vif.cb.penable <= 1; // ACCESS phase @(vif.cb iff vif.pready); // wait for DUT ready vif.cb.psel <= 0; vif.cb.penable <= 0; endtask endclass

// ═══ MONITOR — Silent observer, never drives a single wire ═══ class apb_monitor extends uvm_monitor; `uvm_component_utils(apb_monitor) uvm_analysis_port#(apb_transaction) ap; // broadcast to all subscribers virtual apb_if vif; function void build_phase(uvm_phase phase); ap = new("ap", this); if(!uvm_config_db#(virtual apb_if)::get(this,"","vif",vif)) `uvm_fatal("MON","No virtual interface!") endfunction task run_phase(uvm_phase phase); apb_transaction tr; forever begin // Watch for APB transfer: sel && enable && ready @(vif.cb iff (vif.psel && vif.penable && vif.pready)); tr = apb_transaction::type_id::create("tr"); tr.addr = vif.paddr; tr.write = vif.pwrite; tr.data = vif.pwrite ? vif.pwdata : vif.prdata; tr.strb = vif.pstrb; ap.write(tr); // one write → scoreboard + coverage both get it! `uvm_info("MON", tr.sprint(), UVM_HIGH) end endtask endclass

// ═══ SCOREBOARD — OOP Omar's reference model + checker ═══ class apb_scoreboard extends uvm_scoreboard; `uvm_component_utils(apb_scoreboard) // Array Anna: "associative array = perfect register model!" local bit [31:0] reg_model[bit[11:0]]; // addr → expected data uvm_tlm_analysis_fifo#(apb_transaction) fifo; int unsigned pass_count = 0; int unsigned fail_count = 0; function void build_phase(uvm_phase phase); fifo = uvm_tlm_analysis_fifo#(apb_transaction)::type_id ::create("fifo", this); endfunction task run_phase(uvm_phase phase); apb_transaction tr; forever begin fifo.get(tr); if(tr.write) begin reg_model[tr.addr] = tr.data; // update reference model `uvm_info("SB",$sformatf("WRITE addr=%0h data=%0h",tr.addr,tr.data),UVM_HIGH) end else begin if(reg_model.exists(tr.addr)) begin if(tr.data === reg_model[tr.addr]) begin pass_count++; `uvm_info("SB",$sformatf("PASS read addr=%0h",tr.addr),UVM_HIGH) end else begin fail_count++; `uvm_error("SB",$sformatf("MISMATCH addr=%0h exp=%0h got=%0h", tr.addr, reg_model[tr.addr], tr.data)) end end end end endtask function void report_phase(uvm_phase phase); `uvm_info("SB",$sformatf("RESULTS: PASS=%0d FAIL=%0d",pass_count,fail_count),UVM_NONE) endfunction endclass

// ═══ COVERAGE — Randon Rex tracks what's been tested ═══ class apb_coverage extends uvm_subscriber#(apb_transaction); `uvm_component_utils(apb_coverage) apb_transaction tr; // Randon Rex: "your covergroup is your verification contract!" covergroup apb_cg; // Address space coverage cp_addr: coverpoint tr.addr { bins low_regs = {[12'h000:12'h0FF]}; bins mid_regs = {[12'h100:12'h7FF]}; bins high_regs = {[12'h800:12'hFFF]}; } // Transaction type cp_rw: coverpoint tr.write { bins read_op = {0}; bins write_op = {1}; } // Byte strobes (all combinations tested?) cp_strb: coverpoint tr.strb { bins full_word = {4'hF}; bins upper_half = {4'hC}; bins lower_half = {4'h3}; bins byte_lane = {4'h1,4'h2,4'h4,4'h8}; } // CROSS: addr-range × read/write = 6 bins cx_addr_rw: cross cp_addr, cp_rw; endgroup function new(string n, uvm_component p); super.new(n,p); apb_cg = new(); endfunction function void write(apb_transaction t); tr = t; apb_cg.sample(); endfunction endclass
