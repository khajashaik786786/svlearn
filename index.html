apb_if.sv — Interface + clocking blocks
// APB Interface — Interface Iris's masterpiece interface apb_if(input logic clk, rst_n); logic [11:0] paddr; logic [31:0] pwdata, prdata; logic pwrite, psel, penable, pready, pslverr; logic [3:0] pstrb; // Clocking block for DRIVER — eliminates races! clocking drv_cb @(posedge clk); default input #1step output #1; output paddr, pwdata, pwrite, psel, penable, pstrb; input pready, prdata, pslverr; endclocking clocking mon_cb @(posedge clk); default input #1step; input paddr, pwdata, pwrite, psel, penable, pready, prdata; endclocking modport drv_mp (clocking drv_cb, input rst_n); modport mon_mp (clocking mon_cb, input rst_n); property p_sel_before_en; @(posedge clk) disable iff(!rst_n) penable |-> psel; endproperty assert property(p_sel_before_en) else $error("APB: penable without psel!"); endinterface
apb_transaction.sv — Data class with constraints
class apb_transaction; rand bit [11:0] addr; rand bit [31:0] data; rand bit write; rand bit [3:0] strb; constraint c_addr_align { addr[1:0] == 2'b00; } constraint c_strb_valid { write == 1 -> strb != 4'h0; write == 0 -> strb == 4'hF; } constraint c_write_bias { write dist { 1 := 70, 0 := 30 }; } function string sprint(); return $sformatf("APB: addr=%3h data=%8h %s", addr, data, write?"WR":"RD"); endfunction endclass
apb_driver.sv — Protocol driver via clocking block
class apb_driver; virtual apb_if vif; mailbox#(apb_transaction) gen2drv; task drive_txn(apb_transaction tr); // SETUP phase @(vif.drv_cb); vif.drv_cb.paddr <= tr.addr; vif.drv_cb.pwdata <= tr.data; vif.drv_cb.pwrite <= tr.write; vif.drv_cb.psel <= 1; vif.drv_cb.penable <= 0; // ACCESS phase @(vif.drv_cb); vif.drv_cb.penable <= 1; @(vif.drv_cb iff vif.pready === 1); vif.drv_cb.psel <= 0; vif.drv_cb.penable <= 0; endtask endclass
apb_monitor.sv — Silent observer
class apb_monitor; virtual apb_if vif; mailbox#(apb_transaction) mon2sb; mailbox#(apb_transaction) mon2cov; task run(); apb_transaction tr; forever begin @(vif.mon_cb iff (vif.mon_cb.psel && vif.mon_cb.penable && vif.pready)); tr = new(); tr.addr = vif.mon_cb.paddr; tr.write = vif.mon_cb.pwrite; tr.data = vif.mon_cb.pwrite ? vif.mon_cb.pwdata : vif.prdata; mon2sb.put(tr.copy()); mon2cov.put(tr.copy()); end endtask endclass
apb_scoreboard.sv — Reference model + checker
class apb_scoreboard; bit [31:0] ref_model[bit[11:0]]; int pass_cnt=0, fail_cnt=0; task run(); apb_transaction tr; forever begin mon2sb.get(tr); if (tr.write) begin ref_model[tr.addr] = tr.data; end else begin if (ref_model.exists(tr.addr)) begin if (tr.data === ref_model[tr.addr]) pass_cnt++; else begin fail_cnt++; $error("[SB] FAIL addr=%3h exp=%h got=%h", tr.addr, ref_model[tr.addr], tr.data); end end end end endtask endclass
apb_coverage.sv — Functional coverage
class apb_coverage; apb_transaction curr_tr; covergroup apb_cg; cp_addr: coverpoint curr_tr.addr { bins low_regs = {[12'h000:12'h0FF]}; bins mid_regs = {[12'h100:12'h7FF]}; bins high_regs = {[12'h800:12'hFFF]}; } cp_rw: coverpoint curr_tr.write { bins read_op = {1'b0}; bins write_op = {1'b1}; } cx_addr_rw: cross cp_addr, cp_rw; endgroup task run(); forever begin mon2cov.get(curr_tr); apb_cg.sample(); end endtask endclass
tb_top.sv — Top-level module
module tb_top; logic clk = 0; logic rst_n = 0; always #5 clk = ~clk; // 100 MHz apb_if dut_if(.clk(clk), .rst_n(rst_n)); apb_slave_dut dut( .clk(clk), .rst_n(rst_n), .paddr(dut_if.paddr), .pwdata(dut_if.pwdata), .pwrite(dut_if.pwrite), .psel(dut_if.psel), .penable(dut_if.penable), .pready(dut_if.pready), .prdata(dut_if.prdata) ); apb_env env; initial begin rst_n = 0; repeat(5) @(posedge clk); rst_n = 1; env = new(dut_if, 50); // 50 transactions env.run(); env.report(); $finish; end endmodule
